    private String generateWord() {
        int wordCount = 0;
        ArrayList<String> wordList = new ArrayList<String>();
        try {
            File wordBank = new File("hangwords.txt");
            Scanner cin = new Scanner(wordBank);
            while (cin.hasNextLine()) {
                wordList.add(cin.nextLine());
                wordCount += 1;
            }
            cin.close();
            int randomNum = (int)(Math.random() * wordCount);
            return wordList.get(randomNum).toUpperCase();
        } catch (FileNotFoundException e) {
            System.out.println("An error occurred trying to get hangwords.txt.");
            e.printStackTrace();
            return null;
        }
    }
    private boolean isAllFound() {
        for (JButton b : blankList) {
            if (b.getText().equals("_")) {
                return false;
            }
        }
        return true;
    }
    private void typeLetter(String letter) {
        int index, i; boolean found;
        index = ALPHABET.indexOf(letter);
        if (index >= 0 && clicked[index] == false) {
            clicked[index] = true;
            letterList[index].setBackground(Color.GRAY);
            found = false;
            for (i = 0; i < word.length(); i++) {
                if (letter.equals(word.substring(i, i + 1))) {
                    found = true;
                    blankList[i].setText(letter);
                }
            }
            if (isAllFound()) {
                gameStatus = "won";
                endScreen();
            }
            else if (!found) {
                /* hangman progresses a step */
                errors += 1;
                if (!em) {
                    if (errors == 3 || errors == 5) {
                        errors += 1;
                    }
                }
                hmImage.setIcon(new ImageIcon("hmImages/hmProgression" + errors + ".png"));
                if (errors >= 7) {
                    gameStatus = "lost";
                    endScreen();
                }
            }
        }
    }

    public void resetHangman() {
        int i;
        setVisible(false);
        panel.removeAll();
        panel.validate();
        panel.repaint();
        hmImage.setText("");
        hmImage.setIcon(new ImageIcon("hmImages/hmProgression0.png"));
        panelImg = new JPanel();
        panelImg.setBackground(TRANSPARENT);
        panelImg.setLayout(new GridBagLayout());
        panelImg.add(hmImage);
        panel.add(panelImg);
        /* create blanks for letters to be filled in */
        JPanel panelBlanks = new JPanel();
        word = generateWord();
        blankList = new JButton[word.length()];
        for (i = 0; i < word.length(); i++) {
            JButton btnBlank = new JButton("_");
            btnBlank.setFont(largerFont);
            btnBlank.setForeground(Color.BLACK);
            btnBlank.setBackground(Color.WHITE);
            btnBlank.setOpaque(true);
            btnBlank.setBorderPainted(false);
            btnBlank.addActionListener(this);
            btnBlank.addKeyListener(this);
            blankList[i] = btnBlank;
            panelBlanks.add(blankList[i]);
        }
        panelBlanks.setBackground(TRANSPARENT);
        panel.add(panelBlanks);
        /* create letter keyboard */
        JPanel panelLetters = new JPanel();
        letterList = new JButton[26];
        for (i = 0; i < 26; i++) {
            clicked[i] = false;
            JButton btnLetter = new JButton(ALPHABET.substring(i, i + 1));
            btnLetter.setFont(btnFont);
            btnLetter.setForeground(Color.BLACK);
            btnLetter.setBackground(Color.WHITE);
            btnLetter.setOpaque(true);
            btnLetter.setBorderPainted(false);
            btnLetter.addActionListener(this);
            btnLetter.addKeyListener(this);
            letterList[i] = btnLetter;
            panelLetters.add(letterList[i]);
        }
        panelLetters.setBackground(TRANSPARENT);
        panel.add(panelLetters);
        errors = 0;
        setLayout(null);
        panel.setBackground(TRANSPARENT);
        panel.setLayout(new GridLayout(3, 1));
        panel.setBounds(1, 1, 1100, 750);
        gameStatus = "in progress";
        setVisible(true);
    }
    public void endScreen() {
        setVisible(false);
        panel.removeAll();
        panel.validate();
        panel.repaint();
        panel.addMouseMotionListener(this);
        if (gameStatus.equals("won")) {
            /* won this game */
            /* set hmImage to 'win' version */
            hmImage.setIcon(new ImageIcon("hmImages/freestickman.png"));
            hmImage.setText("<html><br />You win!</html>");
        }
        else {
            /* lost this game */
            /* set hmImage to 'loss' version */
            hmImage.setIcon(new ImageIcon("hmImages/hmProgression7.png"));
            hmImage.setText("<html><br />You lose!</html>");
        }
        /* add hmImage to panelImg to panel */
        panelImg = new JPanel();
        panelImg.setBackground(TRANSPARENT);
        panelImg.setLayout(new GridBagLayout());
        panelImg.add(hmImage);
        panel.add(panelImg);
        /* make new GridBagConstraints so there's space between btns in panelBtns */
        /* add reset btn to panel */
        JButton btnReset = new JButton("Play Again");
        btnReset.setFont(largerFont);
        btnReset.setOpaque(true);
        btnReset.setBorderPainted(false);
        btnReset.setForeground(Color.BLACK);
        btnReset.setBackground(Color.WHITE);
        btnReset.addActionListener(this);
        /* add mode btn to panel */
        JButton btnMode = new JButton("");
        btnMode.setFont(largerFont);
        btnMode.setOpaque(true);
        btnMode.setBorderPainted(false);
        btnMode.setForeground(Color.BLACK);
        if (em) {
            btnMode.setBackground(Color.PINK);

        btnQuit.setFont(largerFont);
        btnQuit.setOpaque(true);
        btnQuit.setBorderPainted(false);
        btnQuit.setForeground(Color.BLACK);
        btnQuit.setBackground(Color.WHITE);
        btnQuit.addActionListener(this);
        /* consolidate control btns */
        JPanel panelBtns = new JPanel();
        panelBtns.setBackground(TRANSPARENT);
        panelBtns.setLayout(new GridBagLayout());
        panelBtns.add(btnReset, c);
        panelBtns.add(btnMode, c);
        panelBtns.add(btnQuit, c);
        panel.add(panelBtns);
        /* set up background */
        setLayout(null);
        panel.setBackground(TRANSPARENT);
        panel.setLayout(new GridLayout(2, 1));
        panel.setBounds(1, 1, 1100, 750);
        setVisible(true);
    }
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() instanceof JButton) {
            /* was not triggered by a Timer event */
            JButton btn = (JButton)e.getSource();
            if (gameStatus.equals("deciding mode")) {
                if (btn.getText().equals(mode1)) {
                    em = true;
                }
                else {
                    em = false;
                }
                gameStatus = "in progress";
                startupHangman(); /* sets up formatting & deals with background */
                resetHangman(); /* makes the actual board */
            }
            else if (gameStatus.equals("won") || gameStatus.equals("lost")) {
                if (btn.getText().equals("Play Again")) {
                    /* resets the game */
                    System.out.println("Solution: \"" + word + "\"");
                    resetHangman();
                    System.out.println("Reset!");
                }
                else if (btn.getText().equals("Quit :(")) {
                    /* quits the game */
                    windowClosing(null);
                }
                else {
                    /* toggles the mode setting */
                    em = !em;
                    if (btn.getBackground().equals(Color.PINK)) {
                        btn.setBackground(Color.CYAN);
                        btn.setText("Use " + mode1 + " Mode");
                        // JOptionPane.showMessageDialog(panel, "Switched to " + mode2 + " Mode!"); /* ask mr. pilon! */

                        // Point position = MouseInfo.getPointerInfo().getLocation();
                        // Point current = position;
                        // while (position.equals(current)) {
                        //     current = MouseInfo.getPointerInfo().getLocation();
                        // }
                        // System.out.println("moved");
                    }
                    else if (btn.getBackground().equals(Color.CYAN)) {
                        btn.setBackground(Color.PINK);
                        btn.setText("Use " + mode2 + " Mode");
                        // JOptionPane.showMessageDialog(panel, "Switched to " + mode1 + " Mode!"); /* ask mr. pilon! */

                        // Point position = MouseInfo.getPointerInfo().getLocation();
                        // Point current = position;
                        // while (position.equals(current)) {
                        //     current = MouseInfo.getPointerInfo().getLocation();
                        // }
                        // System.out.println("moved");
                    }
                }
            }
            else if (gameStatus.equals("in progress")) {
                if (!btn.getText().equals("_")) {
                    /* a letter key was clicked */
                    typeLetter(btn.getText());
                }
            }
            else {
                /* gameStatus isn't any of these! */
                System.out.println("error: gameStatus was unexpected value \"" + gameStatus + "\"");
                windowClosing(null);
            }
        }
        else {
            // System.out.println("timer fired!");
        }
    }

@SuppressWarnings("serial")
class BackgroundPanel extends JPanel {
    private Image bgImage;
    public BackgroundPanel(String imgPath) {
        this(new ImageIcon(imgPath).getImage());
    }
    public BackgroundPanel(Image img) {
        bgImage = img;
        Dimension size = new Dimension(bgImage.getWidth(null), bgImage.getHeight(null));
        setPreferredSize(size);
        setMinimumSize(size);
        setMaximumSize(size);
        setSize(size);
        setLayout(null);
    }
    public void paintComponent(Graphics g) {
        g.drawImage(bgImage, 0, 0, null);
    }
